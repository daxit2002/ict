			members: Data members
			1 - Name
			2 - Age
			3 - Phone number
			4 - Address
			5 - Salary

			classes 'Employee' and 'Manager' inherits the 'Member' class. The 'Employee' and

			manager by making an object of both of these classes and print the same.

        this.phone = phone;
        this.age = age;
    }
    
    void printsalary(){
        System.out.println("Enter employee salary"+salary);
    }
    
}

   String specialization;

}
